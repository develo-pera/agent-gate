2. Create `packages/app/src/app/api/mcp/[tool]/route.ts`:
   ```typescript
   import { NextRequest, NextResponse } from "next/server";
   import { createBridgeContext, toolRegistry } from "@agentgate/mcp-server/src/bridge.js";

   export async function POST(
     request: NextRequest,
     { params }: { params: Promise<{ tool: string }> }
   ) {
     const { tool } = await params;
     const body = await request.json();

     const handler = toolRegistry[tool];
     if (!handler) {
       return NextResponse.json(
         { success: false, error: `Unknown tool: ${tool}` },
         { status: 404 }
       );
     }

     try {
       const walletAddress = body.wallet_address || body.agent_address;
       const ctx = createBridgeContext(walletAddress);
       const result = await handler(body, ctx);
       return NextResponse.json({ success: true, data: result });
     } catch (error) {
       return NextResponse.json(
         { success: false, error: error instanceof Error ? error.message : "Unknown error" },
         { status: 500 }
       );
     }
   }
   ```

   If the import path `@agentgate/mcp-server/src/bridge.js` does not resolve (depends on the mcp-server package.json exports config), use a relative path instead: `../../../../../packages/mcp-server/src/bridge.js` or configure `next.config.ts` transpilePackages. The executor should verify the import resolves and fix the path accordingly.
