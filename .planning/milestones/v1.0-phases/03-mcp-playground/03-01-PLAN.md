export type Domain = "Lido" | "Treasury" | "Delegation" | "ENS" | "Monitor";

  name: string;
  type: ParamType;
  description: string;
  required: boolean;
  default?: string | number | boolean;
  enumValues?: string[];
  isAddress?: boolean;   // Smart default: connected wallet address
  isAmount?: boolean;    // Smart default: "1.0"
}

export interface ToolSchema {

  description: string;     // One-line description from MCP server.tool() call

}
